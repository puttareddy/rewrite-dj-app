  <project_name>AI Agent Platform</project_name>

  <overview>
    The AI Agent Platform is a comprehensive full-stack web application for managing AI agents, their configurations, and monitoring their execution. Users can create and configure AI agents with custom system prompts, model settings, and key-value configurations. The platform tracks all agent executions with detailed metrics including token usage and execution time. It includes role-based access control, API key management for programmatic access, and a modern responsive UI built with Next.js 14, PostgreSQL, Prisma ORM, and Tailwind CSS.
  </overview>

  <database_schema>
    <schema>
      <table name="User">
        <column name="id" type="UUID" primary_key="true"/>
        <column name="email" type="String" unique="true" indexed="true"/>
        <column name="password_hash" type="String"/>
        <column name="name" type="String"/>
        <column name="role" type="Enum(admin, user)"/>
        <column name="created_at" type="DateTime"/>
        <column name="updated_at" type="DateTime"/>
      </table>

      <table name="Agent">
        <column name="id" type="UUID" primary_key="true"/>
        <column name="name" type="String"/>
        <column name="description" type="Text" nullable="true"/>
        <column name="model_type" type="Enum(gpt-4, gpt-3.5-turbo, claude-3, custom)"/>
        <column name="system_prompt" type="Text"/>
        <column name="temperature" type="Decimal" default="0.7"/>
        <column name="max_tokens" type="Integer" default="2048"/>
        <column name="is_active" type="Boolean" default="true"/>
        <column name="user_id" type="UUID" foreign_key="User.id"/>
        <column name="created_at" type="DateTime"/>
        <column name="updated_at" type="DateTime"/>
      </table>

      <table name="AgentExecution">
        <column name="id" type="UUID" primary_key="true"/>
        <column name="agent_id" type="UUID" foreign_key="Agent.id"/>
        <column name="prompt" type="Text"/>
        <column name="response" type="Text"/>
        <column name="tokens_used" type="Integer"/>
        <column name="execution_time_ms" type="Integer"/>
        <column name="status" type="Enum(pending, running, completed, failed)"/>
        <column name="error_message" type="Text" nullable="true"/>
        <column name="created_at" type="DateTime"/>
        <column name="completed_at" type="DateTime" nullable="true"/>
      </table>

      <table name="AgentConfig">
        <column name="id" type="UUID" primary_key="true"/>
        <column name="agent_id" type="UUID" foreign_key="Agent.id"/>
        <column name="key" type="String"/>
        <column name="value" type="String"/>
        <column name="created_at" type="DateTime"/>
        <column name="updated_at" type="DateTime"/>
      </table>

      <table name="APIKey">
        <column name="id" type="UUID" primary_key="true"/>
        <column name="name" type="String"/>
        <column name="key_hash" type="String"/>
        <column name="user_id" type="UUID" foreign_key="User.id"/>
        <column name="is_active" type="Boolean" default="true"/>
        <column name="created_at" type="DateTime"/>
        <column name="last_used_at" type="DateTime" nullable="true"/>
      </table>
    </schema>
  </database_schema>
